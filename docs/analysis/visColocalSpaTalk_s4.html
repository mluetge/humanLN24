plot_st_celltype_mod <- function (object, celltype1, celltype2,
                                  size = 1, color_celltype1 = "blue",
                                  color_celltype2 = "red",
                                  color_others = "grey", alphaOthers = 1) 
{
    if (!is(object, "SpaTalk")) {
        stop("Invalid class for object: must be 'SpaTalk'!")
    }
    st_type <- object@para$st_type
    if_skip_dec_celltype <- object@para$if_skip_dec_celltype
    if (st_type == "single-cell") {
        st_meta <- object@meta$rawmeta
    }
    if (st_type == "spot") {
        if (if_skip_dec_celltype) {
            st_meta <- object@meta$rawmeta
        }
        else {
            st_meta <- object@meta$newmeta
        }
    }
    if (length(celltype1) > 1) {
        stop("Please provide a single cell type name each time!")
    }
    if (length(celltype2) > 1) {
        stop("Please provide a single cell type name each time!")
    }
    if (!celltype1 %in% st_meta$celltype) {
        stop(paste0(celltype1, " is not in the st_meta!"))
    }
    if (is.null(celltype2)) {
      st_meta[!st_meta$celltype %in% c(celltype1), ]$celltype <- "others"
      st_meta$celltype <- factor(st_meta$celltype,
                               levels = c(celltype1, "others"))
      ggplot2::ggplot() + ggplot2::geom_point(ggplot2::aes(st_meta$x, 
        st_meta$y, color = st_meta$celltype), size = size) + 
        ggplot2::scale_color_manual(values = c(color_celltype1, 
            alpha(color_others, alphaOthers))) + ggplot2::theme_bw() +
        ggplot2::theme(panel.grid = ggplot2::element_blank()) + 
        ggplot2::labs(x = "x", y = "y", color = "celltype")

    }else{
          st_meta[!st_meta$celltype %in% c(celltype1, celltype2), ]$celltype <- "others"
          st_meta$celltype <- factor(st_meta$celltype,
                               levels = c(celltype1, celltype2, "others"))
          ggplot2::ggplot() + ggplot2::geom_point(ggplot2::aes(st_meta$x, 
          st_meta$y, color = st_meta$celltype), size = size) + 
          ggplot2::scale_color_manual(values = c(color_celltype1, color_celltype2, 
            alpha(color_others, alphaOthers))) + ggplot2::theme_bw() +
          ggplot2::theme(panel.grid = ggplot2::element_blank()) + 
          ggplot2::labs(x = "x", y = "y", color = "celltype")
    }
}


plot_st_celltype_percent_mod <- function (object, celltype, size = 1,
                                          color_low = NULL, color_mid = NULL, 
                                          color_high = NULL, color_midpoint = NULL,
                                          upper_lim = NULL) 
{
    if (!is(object, "SpaTalk")) {
        stop("Invalid class for object: must be 'SpaTalk'!")
    }
    if_skip_dec_celltype <- object@para$if_skip_dec_celltype
    if (if_skip_dec_celltype) {
        stop("Not availible when skiping dec_celltype()!")
    }
    st_meta <- object@meta$rawmeta
    if (length(celltype) > 1) {
        stop("Please provide a single cell type name each time!")
    }
    if (!celltype %in% colnames(st_meta)) {
        stop(paste0(celltype, " is not in the st_meta!"))
    }
    st_meta$celltype_percent <- as.numeric(st_meta[, celltype])
    if (!is.null(upper_lim)) {
        st_meta$celltype_percent[which(st_meta$celltype_percent > upper_lim)] <- upper_lim
    }
    if (is.null(color_low)) {
        color_low <- "grey"
    }
    if (is.null(color_high)) {
        color_high <- "blue"
    }
    if (is.null(color_mid)) {
        ggplot2::ggplot() + ggplot2::geom_point(ggplot2::aes(st_meta$x, 
            st_meta$y, color = st_meta$celltype_percent), size = size) + 
            ggplot2::scale_color_gradient(low = color_low, high = color_high) + 
            ggplot2::theme_bw() + ggplot2::theme(panel.grid = ggplot2::element_blank()) + 
            ggplot2::labs(x = "x", y = "y", color = substr(celltype, 1, 6) )
    }
    else {
        if (is.null(color_midpoint)) {
            color_midpoint <- stats::median(st_meta$celltype_percent)
        }
        ggplot2::ggplot() + ggplot2::geom_point(ggplot2::aes(st_meta$x, 
            st_meta$y, color = st_meta$celltype_percent), size = size) + 
            ggplot2::scale_color_gradient2(low = color_low, mid = color_mid, 
                high = color_high, midpoint = color_midpoint) + 
            ggplot2::theme_bw() + ggplot2::theme(panel.grid = ggplot2::element_blank()) + 
            ggplot2::labs(x = "x", y = "y", color = substr(celltype, 1, 6) )
    }
}
