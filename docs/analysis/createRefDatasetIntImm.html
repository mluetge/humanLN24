Load reference data and fuse label

basedir <- here()

seuratFRC <- readRDS(file= paste0(basedir,
                            "/data/AllPatWithoutCM_FRConly_intOrig", 
                            "_seurat.rds"))
seuratBEC <- readRDS(file= paste0(basedir,
                            "/data/AllPatWithoutCM_BEConly_intOrig_label", 
                            "_seurat.rds"))
seuratLEC <- readRDS(file= paste0(basedir,
                            "/data/AllPatWithoutCM_LEConly_intOrig_label", 
                            "_seurat.rds"))
seuratIMM <- readRDS(file= paste0(basedir,
                            "/data/AllPatWithoutCM_IMMMerged_integrated_label",
                       "_seurat.rds"))

## add label FRCs
seuratFRC$label <- "medRCIFRC"
seuratFRC$label[which(seuratFRC$intCluster == "7")] <- "BRC"
seuratFRC$label[which(seuratFRC$intCluster == "2")] <- "TRC"
seuratFRC$label[which(seuratFRC$intCluster == "3")] <- "ACTA2+PRC"
seuratFRC$label[which(seuratFRC$intCluster == "4")] <- "VSMC"
seuratFRC$label[which(seuratFRC$intCluster == "5")] <- "PI16+RC"
seuratFRC$label[which(seuratFRC$intCluster == "6")] <- "PRC1"

## group imm cells
seuratIMM$label2 <- seuratIMM$label
seuratIMM$label <- as.character(seuratIMM$label)
seuratIMM$label[which(seuratIMM$label %in% c("naiveCD4-2", "naiveCD4-3",
                                             "naiveCD4-1", "Treg" ))] <- "CD4T"
seuratIMM$label[which(seuratIMM$label %in% c("pDC-2", "pDC-1" ))] <- "pDC"
seuratIMM$label[which(seuratIMM$label %in% c("naiveB", "preB" ))] <- "naiveB"


## group BEC
seuratBEC$label3 <- seuratBEC$label
seuratBEC$label <- "artBEC"
seuratBEC$label[which(seuratBEC$label2 == "Cappilary")] <- "capBEC"
seuratBEC$label[which(seuratBEC$label2 == "Venous")] <- "venBEC"

## group LEC
#seuratLEC$label2 <- seuratLEC$label
seuratLEC$label <- as.character(seuratLEC$label)
seuratLEC <- subset(seuratLEC, label %in% c("SCScLEC", "SCSfLEC", "MedSinusLEC"))
#seuratLEC$label[which(seuratLEC$label2 %in% c("MedCapsLEC", "SCScLEC",
#                                              "ValveLEC"))] <- "cLECcaps"
#seuratLEC$label[which(seuratLEC$label2 %in% c("MedSinusLEC",
#                                              "ParacortLEC"))] <-"MedSinPlusParaLEC"


refSeurat <- merge(seuratFRC, c(seuratIMM, seuratBEC, seuratLEC))
remove(seuratIMM, seuratLEC, seuratBEC, seuratFRC)
refSeurat <- subset(refSeurat, cond == "resting")
refSeurat <- NormalizeData(object = refSeurat)
refSeurat <- FindVariableFeatures(object = refSeurat)
refSeurat <- ScaleData(object = refSeurat, verbose = FALSE)
refSeurat <- RunPCA(object = refSeurat, npcs = 30, verbose = FALSE)
refSeurat <- RunTSNE(object = refSeurat, reduction = "pca", dims = 1:20)
refSeurat <- RunUMAP(object = refSeurat, reduction = "pca", dims = 1:20)

saveRDS(refSeurat, file = paste0(basedir, "/data/refSeurat_allCelltypesIntImm.rds"))
