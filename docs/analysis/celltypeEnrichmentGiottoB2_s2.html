barplot function

## mod from Giotto:::cellProximityBarplot
cellProximityBarplot_mod <- function (gobject, CPscore, min_orig_ints = 5, 
                                      min_sim_ints = 5, p_val = 0.05,
                                      filter_sender = NULL, 
                                      filter_receiver = NULL,
                                      color_by = "type_int", only_pos = F){
    table_mean_results_dc = CPscore$enrichm_res
    original = simulations = p_higher_orig = p_lower_orig = enrichm = type_int = unified_int = NULL
    table_mean_results_dc_filter = table_mean_results_dc[original >= 
        min_orig_ints & simulations >= min_sim_ints]
    table_mean_results_dc_filter = table_mean_results_dc_filter[p_higher_orig <= 
        p_val | p_lower_orig <= p_val]
    table_mean_results_dc_filter <- table_mean_results_dc_filter %>% 
      dplyr::mutate(sender=gsub("--.*", "", unified_int)) %>% 
      dplyr::mutate(receiver=gsub(".*--", "", unified_int))
    if (!is.null(filter_sender)) {
        table_mean_results_dc_filter <- table_mean_results_dc_filter %>% 
          dplyr::filter(sender %in% filter_sender)
    }
    if (!is.null(filter_receiver)) {
        table_mean_results_dc_filter <- table_mean_results_dc_filter %>% 
          dplyr::filter(receiver %in% filter_receiver)
    }
    if (only_pos==TRUE) {
        table_mean_results_dc_filter <- table_mean_results_dc_filter %>% 
          dplyr::filter(enrichm > 0)
    }
     pl <- ggplot2::ggplot()
    pl <- pl + ggplot2::geom_bar(data = table_mean_results_dc_filter, 
        ggplot2::aes(x = unified_int, y = enrichm, fill = eval(parse(text = color_by))), 
        stat = "identity", show.legend = F)
    pl <- pl + ggplot2::coord_flip()
    pl <- pl + ggplot2::theme_bw()
    pl <- pl + ggplot2::labs(y = "enrichment/depletion")
    return(pl)
}
